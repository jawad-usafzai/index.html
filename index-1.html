⚙️ Google Sheets Setup

Connect your form to Google Sheets in 3 easy steps — completely free

1

Create your Google Sheet

Go to sheets.google.com → Create new sheet → Name it exactly: DS Unity Registrations

2

Create a Google Apps Script Web App

  • Open your Google Sheet → click Extensions → Apps Script
  • Delete everything and paste this script:
const SHEET_NAME = "DS Unity Registrations"; const DRIVE_FOLDER = "DS Unity Screenshots"; // optional function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME); // Add headers if sheet is empty if (sheet.getLastRow() === 0) { sheet.appendRow([ "Timestamp","Full Name","Department","Semester", "Phone Number","CNIC","Screenshot URL" ]); } let screenshotUrl = "No screenshot"; // Save screenshot to Drive if provided if (data.screenshot && data.screenshotName) { try { const folders = DriveApp.getFoldersByName(DRIVE_FOLDER); const folder = folders.hasNext() ? folders.next() : DriveApp.createFolder(DRIVE_FOLDER); const decoded = Utilities.base64Decode(data.screenshot); const blob = Utilities.newBlob(decoded, data.screenshotType, data.screenshotName); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); screenshotUrl = file.getUrl(); } catch(err) { screenshotUrl = "Upload failed: " + err.message; } } // Append registration row sheet.appendRow([ new Date().toLocaleString(), data.fullName, data.department, data.semester, data.phone, data.cnic, screenshotUrl ]); return ContentService .createTextOutput(JSON.stringify({ success: true, count: sheet.getLastRow() - 1 })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return ContentService.createTextOutput(JSON.stringify({ count:0, rows:[] })).setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return ContentService.createTextOutput(JSON.stringify({ count:0, rows:[] })).setMimeType(ContentService.MimeType.JSON); const headers = rows[0]; const data = rows.slice(1).map(r => { const obj = {}; headers.forEach((h,i) => obj[h] = r[i]); return obj; }); return ContentService .createTextOutput(JSON.stringify({ count: data.length, rows: data })) .setMimeType(ContentService.MimeType.JSON); }
